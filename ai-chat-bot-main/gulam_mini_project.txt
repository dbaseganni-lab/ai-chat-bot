# Memory to store user inputs
memory = []

# Clean and normalize user input
def clean_input(text):
    return text.strip().lower()

# Match user input to intent or known answer
def match_intent(user_input):
    cleaned = clean_input(user_input)

    if any(word in cleaned for word in greetings):
        return "greeting"
    if any(word in cleaned for word in farewells):
        return "farewell"
    if cleaned in knowledge_base:
        return knowledge_base[cleaned]
    return "default"

# Get a response based on the matched intent
def get_response(intent):
    if intent in responses:
        return random.choice(responses[intent])
    elif intent in knowledge_base.values():
        return intent
    else:
        return random.choice(responses["default"])

# Save user input to memory
def remember(user_input):
    timestamp = datetime.now().strftime("%H:%M:%S")
    memory.append(f"[{timestamp}] {user_input}")
    if len(memory) > 10:
        memory.pop(0)

# Show remembered conversation history
def show_memory():
    print("\n🧠 Conversation Memory:")
    for entry in memory:
        print("•", entry)

# Show help / knowledge base
def show_help():
    print("\n🤖 Try asking me things like:")
    for question in knowledge_base:
        print("•", question.capitalize())

# Main chatbot loop
def main():
    print("="*45)
    print("🤖 Welcome to the Python Chatbot!")
    print("Type something to start chatting.")
    print("Type 'help' for examples or 'quit' to exit.")
    print("="*45)

    while True:
        user_input = input("\nYou: ")
        cleaned = clean_input(user_input)

        if cleaned == "quit":
            show_memory()
            print("\n🤖 Chatbot: It was nice talking to you. Bye!")
            break

        if cleaned == "help":
            show_help()
            continue

        remember(user_input)
        intent = match_intent(cleaned)
        bot_reply = get_response(intent)

        time.sleep(0.5)  # Simulate thinking
        print("🤖 Chatbot:", bot_reply)

if __name__ == "__main__":
    main()
